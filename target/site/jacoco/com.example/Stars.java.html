package com.example;

public class Stars {

    public int countFullStars(int n) {
        int fullStars = 0;

        for (int k = 1; k < n; k++) {
            if (gcd(n, k) == 1) {
                fullStars++;
            }
        }

        return fullStars/2;
    }

    private int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
}
